```javascript
import path from 'path';
import {
  symlink,
  unlink,
  lstat,
  readlink,
  readdir,
  mkdir,
  writeFile,
  readFile,
  rename,
  access,
} from 'fs/promises';

export { symlink, unlink, lstat, readlink, readdir, mkdir, writeFile, readFile, rename, access };

/**
 * Walk a directory recursively and return relative paths of regular files only.
 * Uses lstat semantics — does NOT follow symlinks (Pitfall 7 prevention).
 * @param {string} dirPath - Absolute path to directory to walk
 * @returns {Promise<string[]>} Relative paths of regular files
 */
export async function walkFiles(dirPath) {
  const entries = await readdir(dirPath, { withFileTypes: true, recursive: true });
  return entries
    .filter(e => !e.isDirectory() && !e.isSymbolicLink())
    .map(e => path.relative(dirPath, path.join(e.path, e.name)));
  // e.path is the containing directory (Dirent property, Node 20+)
  // Using e.path (not e.parentPath) for Node 20.12.0 compatibility
}
```

Create `src/config.js` — handles reading and writing the global config at `~/.clawd-linker`.

IMPORTANT: Do NOT use the `conf` library for this. Research Pitfall 8 notes that `conf` stores to `~/Library/Preferences/` by default on macOS, not `~/.clawd-linker`. The config format is trivial (single JSON file), so use plain fs + atomic write pattern instead. The `conf` library remains installed for potential future use but is not used for this config file.

The config file at `~/.clawd-linker` is a JSON file with shape: `{ "repoPath": "/absolute/path", "schemaVersion": 1 }`

```javascript
import path from 'path';
import os from 'os';
import chalk from 'chalk';
import { readFile, writeFile, rename, access } from './utils/fs.js';

export const CONFIG_PATH = path.join(os.homedir(), '.clawd-linker');

/**
 * Read the repo path from ~/.clawd-linker. Exits process with error if:
 * - Config file does not exist (CFG-02)
 * - Config file is not valid JSON (CFG-02)
 * - Configured repo path does not exist on disk (CFG-02)
 * @returns {Promise<string>} Absolute path to the package repository
 */
export async function getRepoPath() {
  let raw;
  try {
    const content = await readFile(CONFIG_PATH, 'utf8');
    raw = JSON.parse(content);
  } catch {
    console.error(chalk.red('No package repository configured. Run `clawd-linker init` first.'));
    process.exit(1);
  }

  if (!raw.repoPath) {
    console.error(chalk.red('Config file is missing repoPath. Run `clawd-linker init` to reconfigure.'));
    process.exit(1);
  }

  try {
    await access(raw.repoPath);
  } catch {
    console.error(chalk.red(`Package repo not found at ${raw.repoPath}. Run \`clawd-linker init\` to reconfigure.`));
    process.exit(1);
  }

  return raw.repoPath;
}

/**
 * Write the repo path to ~/.clawd-linker using atomic write (tmp + rename).
 * @param {string} repoPath - Absolute path to the package repository
 */
export async function setRepoPath(repoPath) {
  const resolved = path.resolve(repoPath);
  const data = JSON.stringify({ schemaVersion: 1, repoPath: resolved }, null, 2);
  const tmp = CONFIG_PATH + '.tmp';
  await writeFile(tmp, data, 'utf8');
  await rename(tmp, CONFIG_PATH);
}
```

Key design decisions per research:
- Uses atomic write pattern (write to .tmp, then rename) per Pitfall 4
- Uses `path.resolve()` to ensure absolute paths per Pitfall 2
- Imports from `./utils/fs.js` — never directly from `fs/promises`
- Exits with chalk.red error messages for all failure cases (CFG-02)
